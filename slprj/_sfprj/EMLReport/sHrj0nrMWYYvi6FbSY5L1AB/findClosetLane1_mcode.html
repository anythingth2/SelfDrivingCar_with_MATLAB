1function [LeftClosetPts,RightClosetPts] = findClosetLane(LinePts,Slope,SIZE_ROI_X,SIZE_ROI_Y)
 2    
 3    closetLeft = int32(9999);
 4    closetRight = int32(9999);
 5    for i=1:5
 6        x = ( 0 - SIZE_ROI_Y - LinePts(i,2) )/Slope(i) + LinePts(i,1);
 7        leftDiff = SIZE_ROI_X/2 - x;
 8        rightDiff = x - SIZE_ROI_X;
 9        if closetLeft > leftDiff(1)
10            closetLeft = leftDiff(1);
11            LeftClosetPts = LinePts(i);
12        end
13        
14        if closetRight > rightDiff(1)
15            closetRight = rightDiff(1);
16            RightClosetPts = LinePts(i);
17        end
18    end
