1function [LeftClosetPts,RightClosetPts] = findClosetLane(LinePts,Slope,SIZE_ROI_X,SIZE_ROI_Y)
 2    LeftClosetPts = int32([0 0 0 0]);
 3    RightClosetPts = int32([0 0 0 0]);
 4    closetLeft = int32(9999);
 5    closetRight = int32(9999);
 6    for i=1:5
 7        x = ( 0 - SIZE_ROI_Y - LinePts(i,2) )/Slope(i) + LinePts(i,1);
 8        leftDiff = SIZE_ROI_X/2 - x;
 9        rightDiff = x - SIZE_ROI_X;
10        if closetLeft > leftDiff(1)
11            closetLeft = leftDiff(1);
12            LeftClosetPts = LinePts(i);
13        end
14        
15        if closetRight > rightDiff(1)
16            closetRight = rightDiff(1);
17            RightClosetPts = LinePts(i);
18        end
19    end
