8static void TrackNonRecursive_UR ( boolean_T *edge, 



9                                real32_T  *tmpOrMag,



10                                int16_T   *stackR,



11                                int16_T   *stackC,



12                                int16_T   *stackP,



13                                int16_T   *stackQ,



14                                int_T      r, 



15                                int_T      c, 



16                                real32_T     lowTh, 



17                                int_T      inpRows, 



18                                int_T      inpCols



19                               )



20{



21    int_T p,q,R,C,P,Q;



22    int_T stackIdx            = -1;



23    int_T stackMaxLength      = (inpRows*inpCols);



24    int_T init                = 1;



25    boolean_T prevIsStackDecr = 0;



26    boolean_T IsStackIncr     = 1; 



27    boolean_T IsStackDecr     = 0; 



28    



29   while ((stackIdx>=0 && stackIdx < stackMaxLength) || init)



30   {  



31   /* stackIdx < stackMaxLength this avoids stack overflow;           */



32   /* if stack goes beyond this maxLength, the routine quits and this */



33   /* will give wrong result */



34



35    /*********** init and stack increment stage **********/



36    if (IsStackIncr)



37    {



38        R = init? r : stackR[stackIdx] + stackP[stackIdx]; /* Iold+p; */



39        C = init? c : stackC[stackIdx] + stackQ[stackIdx]; /* Jold+q; */



40



41        IsStackIncr = 0;



42



43        if (edge[R+C*inpRows] == 0) 



44        {



45            edge[R+C*inpRows] = 1;  



46                



47            for (p= -1; p<=1; p++)  



48            {



49                for(q= -1; q<=1; q++)   



50                {



51                    if (p==0 && q==0) continue;



52                    if (isInRange(R+p, C+q, inpRows, inpCols) && 



53                        tmpOrMag[(R+p)+(C+q)*inpRows] >= lowTh)



54                    {



55                        stackIdx++;



56                        stackR[stackIdx] = (int16_T)R; /* Iold+p; */



57                        stackC[stackIdx] = (int16_T)C; /* Jold+q; */



58



59                        stackP[stackIdx] = (int16_T)p; 



60                        stackQ[stackIdx] = (int16_T)q; 



61



62                        IsStackIncr=1;  



63                        prevIsStackDecr=0;



64                            



65                        break;



66                    }



67                }  



68                if (IsStackIncr) break;   



69            } 



70        }



71        if  (init)



72           init=0;



73        else



74           if (IsStackIncr==0) IsStackDecr=1;



75    }



76    /*********** stack decrement stage **********/



77    else if (IsStackDecr)



78    {



79        int_T skip_num_iter, count=0;  



80



81        R = stackR[stackIdx]; 



82        C = stackC[stackIdx]; 



83        P = stackP[stackIdx];



84        Q = stackQ[stackIdx];



85



86        skip_num_iter = 3*(P+1)+ (Q+1);



87        for (p= -1; p<=1; p++)  



88        {



89            for(q= -1; q<=1; q++,count++)   



90            {



91                if (count>skip_num_iter)



92                {



93                    if (p==0 && q==0) {continue;}



94                    if (!prevIsStackDecr && isInRange(R+p, C+q, inpRows, inpCols) && 



95                        tmpOrMag[(R+p)+(C+q)*inpRows] >= lowTh)



96                    {



97                        stackR[stackIdx] = (int16_T)R; /* R+p; */



98                        stackC[stackIdx] = (int16_T)C; /* C+q; */



99



100                        stackP[stackIdx] = (int16_T)p; 



101                        stackQ[stackIdx] = (int16_T)q; 



102



103                        IsStackDecr=0;  



104                        IsStackIncr=1;



105                        prevIsStackDecr=0;



106



107                        break;



108                    }



109                }



110



111            } 



112            if (IsStackIncr) break; 



113        } 



114        if (IsStackDecr) 



115        {



116            stackIdx--;



117            prevIsStackDecr=1;



118        }



119    }



120



121   }



122}

124static void HysteresisThresholding_UR (boolean_T *edge, 



125                                    real32_T  *tmpOrMag,



126                                    int16_T   *stackR,



127                                    int16_T   *stackC,



128                                    int16_T   *stackP,



129                                    int16_T   *stackQ,



130                                    real32_T   highTh, 



131                                    real32_T   lowTh, 



132                                    int_T      inpRows, 



133                                    int_T      inpCols)



134{



135    /* Hysteresis is used to track along the remaining pixels that have not      */



136    /* been suppressed. Hysteresis uses two thresholds and if the magnitude      */



137    /* is below the first threshold, it is set to zero (made a nonedge). If      */



138    /* the magnitude is above the high threshold, it is made an edge. And if     */



139    /* the magnitude is between the 2 thresholds, then it is set to zero unless  */



140    /* there is a path from this pixel to a pixel with a gradient above T2.      */



141



142    int_T r,c;



143    



144    memset(edge,0,inpRows*inpCols*sizeof(boolean_T));



145



146    for (r=0; r<inpRows; r++)



147      for (c=0; c<inpCols; c++)



148        if (tmpOrMag[r+c*inpRows] >= highTh)



149             TrackNonRecursive_UR (edge,tmpOrMag,stackR,stackC,stackP,stackQ,



150                                r, c, lowTh, inpRows,inpCols);



151}

154LIBMWVISIONRT_API void MWVIP_EdgeCanny_userTh_R(



155    const real32_T  *inpImg,



156    const real32_T  *gauss1D,



157    const real32_T  *dgauss1D,



158          real32_T  *cFiltered,  /* DWork same size as image */



159          real32_T  *rFiltered,  /* DWork same size as image */



160         boolean_T  *outEdge,



161          real32_T  *tmpOrMag,   /* DWork same size as image */



162    const real32_T  *ThreshCanny,



163          int_T  inpRows,



164          int_T  inpCols,



165          int_T  halfFiltLen)



166{



167    real32_T lowTh  = ThreshCanny[0]; 



168    real32_T highTh = ThreshCanny[1];



169    int16_T *stackR;



170    int16_T *stackC;



171    int16_T *stackP;



172    int16_T *stackQ;



173    const int_T quarterFiltLen = halfFiltLen/2;



174    const int_T inpWidth       = inpRows*inpCols;



175    int_T r,c;



176    



177    /* step-1: filter input image along row and along column with Gaussian filter */



178    MWVIP_RC_Gaussian_Smoothing_R(inpImg,gauss1D,cFiltered,rFiltered,inpRows,inpCols,halfFiltLen);



179



180    /* step-2: filter smoothed image along column with the derivative of Gaussian filter */



181    MWVIP_C_Derivative_Image_R(cFiltered,dgauss1D,tmpOrMag,inpRows,inpCols,halfFiltLen);



182    memcpy((byte_T *)cFiltered,(byte_T *)tmpOrMag,inpWidth*sizeof(real32_T));



183



184    /* step-3: filter smoothed image along row with the derivative of Gaussian filter */



185    MWVIP_R_Derivative_Image_R(rFiltered,dgauss1D,tmpOrMag,inpRows,inpCols,halfFiltLen);



186    memcpy((byte_T *)rFiltered,(byte_T *)tmpOrMag,inpWidth*sizeof(real32_T));



187



188    /* step-4: get the magnitude of cFilt and rFilt with non-maximum suppression */



189    MWVIP_NonMaximum_Suppression_R(cFiltered,rFiltered,tmpOrMag,inpRows,inpCols); 



190



191    /* step-5: Hysteresis thresholding of edge pixels */



192    stackR = (int16_T *)cFiltered;



193    stackC = (int16_T *)rFiltered;



194    stackP = &stackR[inpWidth];



195    stackQ = &stackC[inpWidth];



196



197    HysteresisThresholding_UR (outEdge,tmpOrMag,stackR,stackC,stackP,stackQ,



198                            highTh,lowTh,inpRows,inpCols);



199



200    /* step-6: Take care of border pixels */



201    for (r=0; r<quarterFiltLen; r++)



202      for (c=0; c<inpCols; c++)



203        outEdge[r+c*inpRows] = 0; 



204



205    for (r=inpRows-1; r>inpRows-1-quarterFiltLen; r--)



206      for (c=0; c<inpCols; c++)



207        outEdge[r+c*inpRows] = 0; 



208



209    for (r=0; r<inpRows; r++)



210      for (c=0; c<quarterFiltLen; c++)



211        outEdge[r+c*inpRows] = 0; 



212



213    for (r=0; r<inpRows; r++)



214      for (c=inpCols-quarterFiltLen-1; c<inpCols; c++)



215        outEdge[r+c*inpRows] = 0; 



216}
