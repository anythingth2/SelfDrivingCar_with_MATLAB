14LIBMWVISIONRT_API void MWVIP_DrawText_RGB_double_AA(const uint8_T* bitmap,



15                                             int32_T pen_x,



16                                             int32_T pen_y,



17                                             int32_T left_bearing,



18                                             int32_T top_bearing,



19                                             uint16_T bitmapWidth,



20                                             uint16_T bitmapHeight,



21                                             uint32_T imageWidth,



22                                             uint32_T imageHeight,



23                                             void* outputImageR,



24                                             void* outputImageG,



25                                             void* outputImageB,



26                                             const void* colorVect,



27                                             const void* opacityPtr,



28                                             boolean_T isImageTransposed)



29{ 



30    int32_T  i, j,x,y;



31    uint32_T pixelIndexCpy;



32    real_T	bitmapVal, valR, valG, valB;



33    real_T *outR = (real_T*) outputImageR,



34        *outG = (real_T*) outputImageG,



35        *outB = (real_T*) outputImageB;



36    const real_T*  colorVector = (const real_T*) colorVect; 



37    real_T opacity = *((const real_T*)opacityPtr);



38    int_T outerLoop, innerLoop, bitmapIndx = 0;



39    if (isImageTransposed) {



40        outerLoop = bitmapHeight;



41        innerLoop = bitmapWidth;



42        y = pen_x + left_bearing;



43        x = pen_y - top_bearing;



44    } else {



45        outerLoop = bitmapWidth;



46        innerLoop = bitmapHeight; 



47        x = pen_x + left_bearing;



48        y = pen_y - top_bearing;



49    }



50    pixelIndexCpy = y + (x * imageHeight);



51    for(i = 0; i < outerLoop; i++) {



52        uint32_T pixelIndex = pixelIndexCpy - 1;



53        for(j = 0; j < innerLoop; j++)



54        {



55            pixelIndex++;



56            if(	(x < (-i)) || (x >= ((int32_T)imageWidth-i)) ||



57                (y < (-j)) || (y >= ((int32_T)imageHeight-j))) {



58                    bitmapIndx++;



59                    continue;



60            }



61



62            if (isImageTransposed)



63                bitmapVal = (real_T)bitmap[bitmapIndx++] / 255.0;



64            else



65                bitmapVal = (real_T)bitmap[j * bitmapWidth + i] / 255.0;



66            



67



68            if(bitmapVal == 0.0)



69                continue;



70



71            if(bitmapVal == 1.0)	/* just the text color  */



72            {



73                valR = colorVector[0];



74                valG = colorVector[1];



75                valB = colorVector[2];



76            }



77            else	/* on the edge, need to figure out for nice anti-aliasing */



78            {



79                valR = (colorVector[0] - outR[pixelIndex]) * bitmapVal + outR[pixelIndex];



80                valG = (colorVector[1] - outG[pixelIndex]) * bitmapVal + outG[pixelIndex];



81                valB = (colorVector[2] - outB[pixelIndex]) * bitmapVal + outB[pixelIndex];



82            }



83



84            if(opacity < 1.0) 



85            {



86                valR = (opacity * valR) + ((1.0 - opacity) * outR[pixelIndex]);



87                valG = (opacity * valG) + ((1.0 - opacity) * outG[pixelIndex]);



88                valB = (opacity * valB) + ((1.0 - opacity) * outB[pixelIndex]);



89            }



90



91            outR[pixelIndex] = valR;



92            outG[pixelIndex] = valG;



93            outB[pixelIndex] = valB;



94        }



95        pixelIndexCpy += imageHeight;



96    }



97}
